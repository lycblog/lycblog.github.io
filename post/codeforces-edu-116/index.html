题解

A. AB Balance

思路

连续 a 段和连续 b 段的交界处会使 ABABAB 子串和 BABABA 子串的数量增加。如果输入的字符串中首尾两个字母相同，就不需要调整。否则只需反转这个字符串的第一个字母或最后一个字母即可。

代码

#include <iostream>
#include <string>
using namespace std;
inline char rev(char c) {
	if (c == 'a') return 'b';
	return 'a';
}
int main() {
	int T; cin >> T;
	while (T--) {
		string s; cin >> s;
		int cnta = 0, cntb = 0;
		for (int i = 0; i < s.size(); i++) {
			if (s[i] == 'a' && (i == 0 || s[i - 1] != 'a')) cnta++;
			else if (s[i] == 'b' && (i == 0 || s[i - 1] != 'b')) cntb++;
		}
		if (cnta == cntb) {
			s[0] = s[s.size() - 1];
		}
		cout << s << '\n';	
	}
	return 0;
}

B. Update Files

思路

假如我们有无穷多条电缆，在第 iii 小时有 2i2^i2i 台电脑接收到文件。由于电缆数量是有限的，如果 2i>k2^i>k2i>k，则在第 iii 小时有 kkk 台电脑接收到文件。

由于 111 号电脑在一开始就已经安装文件了，当 n=1n=1n=1 时，输出结果为 000。否则我们把 nnn 减 111。此时，nnn 表示还未安装更新文件的电脑数量。

我们先考虑 2i≤k2^i \le k2i≤k 的情况。每次循环时，先把 nnn 减去 2i2^i2i。如果 n≤0n \le 0n≤0，就表示所有电脑都已经接收到文件，输出 iii。

如果执行完上述过程，仍有电脑没接收到文件，输出执行上述循环的次数，加上 n / k + min(1ll, n % k) 的值。

代码

#include <iostream>
#define int long long
using namespace std;
signed main() {
	int T; cin >> T;
	while (T--) {
		int n, k; cin >> n >> k;
		if (n == 1) {
			cout << "0\n";
			continue;
		}
		n--;
		bool flag = false;
		int cnt = 0;
		for (int i = 1; i <= k; i *= 2) {
			n -= i, cnt++;
			if (n <= 0) {
				cout << cnt << '\n';
				break; 
			}
		}
		if (n > 0) {
			cout << cnt + n / k + min(1ll, n % k) << '\n';
		}
	} 
	return 0;
}

C. Banknotes

思路

如果要使得 f(s)>kf(s) > kf(s)>k，我们需要使 f(s)=k+1f(s)=k+1f(s)=k+1。所以我们要先 k++ 一下。

从前往后看 aaa 数组，计算相邻两位的差 did_idi​。如果此时 kkk 的数值小于 10di−110^{d_i}-110di​−1，则 bi=kb_i=kbi​=k，否则 bi=10di−1b_i=10^{d_i}-1bi​=10di​−1。

最后倒序输出 bib_ibi​ 即可。注意去掉前导 000。

代码

#include <iostream>
using namespace std;
int a[20], b[20], p[10];
int main() {
	int T; cin >> T;
	p[0] = 1;
	for (int i = 1; i <= 9; i++) p[i] = p[i - 1] * 10;
	while (T--) {
		for (int i = 0; i < 20; i++) b[i] = 0;
		int n, k; cin >> n >> k;
		k++;
		for (int i = 1; i <= n; i++) cin >> a[i];
		for (int i = 1; i < n; i++) {
			int d = a[i + 1] - a[i];
			b[i] = min(p[d] - 1, k);
			k -= b[i];
			if (!k) break;
		}
		if (k) b[n] = k;
		bool out = false;
		for (int i = n; i >= 1; i--) {
			if (!out && b[i] > 0) out = true;
			if (out) cout << b[i];
		}
		if (!out) cout << 0;
		cout << '\n';
	}
	return 0;
}

D. Red-Blue Matrix

思路

我们可以按第一列元素的值将每行进行排序，并记录每行初始的位置。这样，我们可以将矩阵中要填成红色的部分和要填成蓝色的部分分离开。

这样，我们可以把矩阵分成四个部分：

对于在第 jjj 列，红、蓝区分界线两侧的元素 ai,ja_{i,j}ai,j​ 和 ai+1,ja_{i+1,j}ai+1,j​，如果 AAA 区前缀最大值 Ai,jA_{i,j}Ai,j​ 小于 CCC 区前缀最小值 Ci+1,jC_{i + 1,j}Ci+1,j​，那么我们令 okli,j=1okl_{i,j}=1okli,j​=1，否则我们令 okli,j=0okl{i,j}=0okli,j=0；如果 DDD 区前缀最大值 Di+1,jD_{i+1,j}Di+1,j​ 小于 BBB 区前缀最小值 Bi,jB_{i,j}Bi,j​，那么我们令 okri,j=1okr_{i,j}=1okri,j​=1，否则我们令 okri,j=0okr{i,j}=0okri,j=0。

然后我们枚举横向、纵向两条分界线。若分界线左上角元素的下标为 i,ji,ji,j，若满足 okl[i][j]=1okl[i][j]=1okl[i][j]=1 且 okr[i][j+1]okr[i][j + 1]okr[i][j+1]，我们输出 YES 以及方案。由于答案不唯一，只需输出一种方案即可，因此输出这种方案后不必再找后面的下标。如果所有下标都不满足这个条件，输出 NO。

由于排序的时间复杂度为 O(nlog⁡n)O(n \log n)O(nlogn)，该算法整体时间复杂度为 O(nlog⁡n+nm)O(n \log n+nm)O(nlogn+nm)

代码

#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
struct arr {
	int pos; vector<int> t;
	int &operator[](int x) {return t[x];} //像数组下标一样访问 
};
vector<arr> a, A, B, C, D, okl, okr;
bool cmp(arr aa, arr bb) {return aa[1] < bb[1];}
int p[500005];
char ans[500005];
int main() {
	int T; cin >> T;
	while (T--) {
		int n, m; cin >> n >> m;
		a.clear();
		A.clear();
		B.clear();
		C.clear();
		D.clear();
		okl.clear();
		okr.clear();
		a.resize(n + 1);
		A.resize(n + 1);
		B.resize(n + 1);
		C.resize(n + 1);
		D.resize(n + 1);
		okl.resize(n + 1);
		okr.resize(n + 1);
		for (int i = 1; i <= n; i++) {
			a[i].t.clear();
			A[i].t.clear();
			B[i].t.clear();
			C[i].t.clear();
			D[i].t.clear();
			okl[i].t.clear();
			okr[i].t.clear();
			a[i].t.resize(m + 1);
			A[i].t.resize(m + 1);
			B[i].t.resize(m + 1);
			C[i].t.resize(m + 1);
			D[i].t.resize(m + 1);
			okl[i].t.resize(m + 1);
			okr[i].t.resize(m + 1);
			for (int j = 1; j <= m; j++) {
				cin >> a[i][j];
				a[i].pos = i;
			}
		}
		sort(a.begin() + 1, a.begin() + n + 1, cmp);
		for (int j = 1; j <= m; j++) {
			A[1][j] = B[1][j] = a[1][j];
			C[n][j] = D[n][j] = a[n][j];
		}
		for (int j = 1; j <= m; j++) {
			for (int i = 2; i <= n; i++) {
				A[i][j] = max(A[i - 1][j], a[i][j]);
				B[i][j] = min(B[i - 1][j], a[i][j]); 
			}
			for (int i = n - 1; i >= 1; i--) {
				C[i][j] = min(C[i + 1][j], a[i][j]);
				D[i][j] = max(D[i + 1][j], a[i][j]);
			}
		}
		for (int i = 1; i <= n; i++) {
			for (int j = 2; j <= m; j++) {
				A[i][j] = max(A[i][j - 1], A[i][j]);
				C[i][j] = min(C[i][j - 1], C[i][j]);
			}
			for (int j = m - 1; j >= 1; j--) {
				B[i][j] = min(B[i][j + 1], B[i][j]);
				D[i][j] = max(D[i][j + 1], D[i][j]);
			}
		} 
		for (int j = 1; j <= m; j++) {
			for (int i = 1; i < n; i++) {
				okl[i][j] = (A[i][j] < C[i + 1][j]);
				okr[i][j] = (D[i + 1][j] < B[i][j]);
			}
		}
		ans[n] = '\0'; //直接输出字符串时，一定不要忘了在字符串最后加终止符！
		bool ok = false;
		for (int i = 1; i < n; i++) {
			ans[a[i].pos - 1] = 'B';
			for (int j = 1; j < m; j++) {
				if (okl[i][j] && okr[i][j + 1]) {
					p[i] = j;
					ok = true;
					cout << "YES\n";
					for (int k = i + 1; k <= n; k++) ans[a[k].pos - 1] = 'R';
					cout << ans << ' ' << j << '\n';
					break;
				}
			}
			if (ok) break;
		}
		if (!ok) cout << "NO\n";
	}
	return 0;
}
